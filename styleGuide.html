Navigation Button

<a class="button-nav" href="#">Default Button</a>

Payment Form

Sample code:

          
<form class="payment-form">
  <fieldset id="customer-info">
    <legend>Delivery Info</legend>
    <label for="payment-name">Name:</label>
    <div class="payment-container" id="payment-name">
      <input type="text" class="payment-text two-split" id="first-name" placeholder="First Name">
      <input type="text" class="payment-text two-split" id="last-name" placeholder="Last Name">
    </div>
    <label for="address" >Address:</label>
    <input type="text" class="payment-text" id="address" placeholder="123 Example St">
    <label for="country" >Country:</label>
    <input type="text" class="payment-text" id="country">
    <label for="city" >City:</label>
    <input type="text" class="payment-text" id="city">
    <label for="province" >Province:</label>
    <input type="text" class="payment-text" id="province">
    <label for="postal-code" >Postal Code:</label>
    <input type="text" class="payment-text" id="postal-code">
    <label for="email" >Email:</label>
    <input type="text" class="payment-text" id="email" placeholder="example@email.com">

  </fieldset>

  <fieldset id="shipping-options">
    <legend>Shipping Options</legend>
    <input type="radio" id="standard" name="shipping" value="standard">
    <label for="standard">Standard Shipping</label>

    <input type="radio" id="express" name="shipping" value="express">
    <label for="express">Express Shipping</label>
  </fieldset>

  <button type="submit">Submit</button>
</form>
